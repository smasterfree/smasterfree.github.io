需求

使用尽可能少的计算节点资源，模拟大规模测试环境。计算节点和网关之间管控面主要发送的是lldp in vxlan的管控心跳报文。

因此考虑写一个fake-agent来模拟计算节点发送的心跳报文，测试网关处理心跳报文能力，以及在此背景下重启、批量增删agent有无问题。

func buildLLDPPayload(ip string) string {

	hexNow := time2hex()
	typeLtime := "fe0c02000001"
	typeEtime := "fe0c02000002"
	typeAgent := "fe0802000003"
	typeEnd := "0000"
	agentip := ip2hex(ip)

	pkts := typeLtime + hexNow + typeEtime + hexNow + typeAgent + agentip + typeEnd
	return pkts
}
